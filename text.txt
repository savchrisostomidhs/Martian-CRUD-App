# Create resource
$body = @{name="Laptop"; id=1; quantity=5} | ConvertTo-Json
Invoke-RestMethod -Uri "http://localhost:3000/resources" -Method Post -Body $body -ContentType "application/json"

# Get resource
Invoke-RestMethod -Uri "http://localhost:3000/resources/1" -Method Get

# Update
$update = @{quantity=10} | ConvertTo-Json
Invoke-RestMethod -Uri "http://localhost:3000/resources/1" -Method Patch -Body $update -ContentType "application/json"

# Delete
Invoke-RestMethod -Uri "http://localhost:3000/resources/1" -Method Delete
